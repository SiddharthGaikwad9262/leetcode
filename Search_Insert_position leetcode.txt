// Search_Insert_position leetcode

class Solution {
    public int searchInsert(int[] nums, int target) {
        int i;
        for( i=0;i<nums.length;i++)
        {
            if(nums[i]==target)
            {
                return i;
            }
            else if (i!=nums.length-1 &&( nums[i]<target&&nums[i+1]>target))
            {
                return i+1;
            }
            else if (nums[i]>target)
            {
                return i;
            }
           
        }

        return i;
       
    }
}
